  def cmd_quality_audit(domain: str = typer.Option("healthcare", "--domain", "-d", ...),
                         as_json: bool = typer.Option(False, "--json", ...)) -> None:
      from knowledge_lake.pipeline.quality_audit import run_quality_audit
